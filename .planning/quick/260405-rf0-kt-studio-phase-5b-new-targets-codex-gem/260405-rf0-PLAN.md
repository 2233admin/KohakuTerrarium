From base.py -- Target ABC (every target must implement):
```python
class Target(ABC):
    name: str          # registry key
    display_name: str  # human label

    @abstractmethod
    def detect(self) -> Path | None: ...
    @abstractmethod
    def build_command(self, profile: ProfileConfig, settings_path: Path | None = None) -> list[str]: ...
    @abstractmethod
    def list_models(self) -> list[str]: ...
    @abstractmethod
    def status(self) -> dict: ...

    # Optional with defaults:
    def settings_path(self) -> Path | None: return None
    def merge_settings(self, base: dict, profile: ProfileConfig) -> dict: return base
    def session_dir(self) -> Path | None: return None
    def scan_sessions(self) -> list[dict]: return []
```

    ```python
    def scan_all_sessions() -> list[dict]:
        """Aggregate sessions from all installed targets."""
        from kohakuterrarium.studio.targets import list_targets
        all_sessions: list[dict] = []
        for target in list_targets():
            if target.detect() is None:
                continue
            try:
                sessions = target.scan_sessions()
                for s in sessions:
                    s["target"] = target.name
                all_sessions.append(s) if not sessions else all_sessions.extend(sessions)
            except Exception:
                logger.debug("Failed to scan sessions for %s", target.name)
        all_sessions.sort(key=lambda s: s.get("modified", ""), reverse=True)
        return all_sessions
    ```
